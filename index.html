}); } const t = sheet_(TEMPLATES_SHEET); if (t && t.getLastRow() > 1) { t.getRange(2, 1, t.getLastRow() - 1, 5).getValues().forEach(r => { if (r[0] !== '' && r[0] !== null) templates.push({ id: Number(r[0]) || r[0], name: String(r[1] || ''), type: String(r[2] || '').toLowerCase(), exercises: String(r[3] || ''), duration: Number(r[4]) || 0 }); }); } const p = sheet_(PLAN_SHEET); if (p && p.getLastRow() > 1) { p.getRange(2, 1, p.getLastRow() - 1, 5).getValues().forEach(r => { if (r[0] !== '' && r[0] !== null) plan.push({ date: dateStr_(r[0]), name: String(r[1] || ''), type: String(r[2] || '').toLowerCase(), exercises: String(r[3] || ''), duration: Number(r[4]) || 0 }); }); } return json_({ sessions, weights, templates, plan }); } // ── Sessions ───────────────────────────────────────────────────────────────── function addSession_(s) { const sh = sheet_(SESSIONS_SHEET); return json_({ ok: true }); } return json_({ error: 'Session not found.' }); } function deleteSession_(id) { const sh = sheet_(SESSIONS_SHEET); if (!sh) return json_({ error: 'Sessions sheet not found.' }); const data = sh.getDataRange().getValues(); for (let i = 1; i < data.length; i++) if (String(data[i][0]) === String(id)) { sh.deleteRow(i + 1); return json_({ ok: true }); } return json_({ error: 'Session not found.' }); } // ── Weights ────────────────────────────────────────────────────────────────── function addWeight_(w) { const sh = sheet_(WEIGHTS_SHEET); if (!sh) return json_({ error: 'Weights sheet not found.' });
